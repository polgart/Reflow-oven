103 void UART2_Initialize(void)
104 {
105  IEC1bits.U2TXIE = 0;

108  // STSEL 1; IREN disabled; PDSEL 8N; UARTEN enabled; RTSMD disabled; USIDL disabled; WAKE disabled; ABAUD disabled; LPBACK disabled; BRGH enabled; URXINV disabled; UEN TX_RX;
109  // Data Bits = 8; Parity = None; Stop Bits = 1;
110  U2MODE = (0x8008 & ~(1<<15)); // disabling UART ON bit
111  // UTXISEL0 TX_ONE_CHAR; UTXINV disabled; URXEN disabled; OERR NO_ERROR_cleared; URXISEL RX_ONE_CHAR; UTXBRK COMPLETED; UTXEN disabled; ADDEN disabled;
112  U2STA = 0x00;
113  // BaudRate = 9600; Frequency = 4000000 Hz; U2BRG 103;
114  U2BRG = 0x67;
115  // ADMADDR 0; ADMMASK 0;
116  U2ADMD = 0x00;

119  txTail = txQueue;
120  rxHead = rxQueue;
121  rxTail = rxQueue;
122 
123  rxOverflowed = false;

131  //Make sure to set LAT bit corresponding to TxPin as high before UART initialization

199 void __attribute__ ( ( interrupt, no_auto_psv ) ) _U2RXInterrupt( void )
200 {
201  (*UART2_RxDefaultInterruptHandler)();
202 
203  IFS1bits.U2RXIF = 0;
204 
205  while((U2STAbits.URXDA == 1))
206  {
207  *rxTail = U2RXREG;
208 
209  // Will the increment not result in a wrap and not result in a pure collision?
210  // This is most often condition so check first
211  if ( ( rxTail != (rxQueue + UART2_CONFIG_RX_BYTEQ_LENGTH-1)) &&
212  ((rxTail+1) != rxHead) )
213  {
214  rxTail++;
215  }
216  else if ( (rxTail == (rxQueue + UART2_CONFIG_RX_BYTEQ_LENGTH-1)) &&
217  (rxHead != rxQueue) )
218  {
219  // Pure wrap no collision
220  rxTail = rxQueue;
221  }
222  else // must be collision
223  {
224  rxOverflowed = true;
225  }
226  }
227 }
