650 void baseSW_TMR2_ISR(void) {
651 
652  // SSR control
653  if (HEAT_IN_PROGRESS) {
654  if (checkStartConditions()) {
655  // Increase timeStamp by 100 ms
656  uint16_t cTimeStamp = heatTimerObj.timeStamp++;
657 
658  //Get current temperature offset
659  uint16_t cOffset = temperatureHeatProfile.currentProfile.offset;
660 
661  //Get the number of records in the heat profile
662  uint16_t mOffset = temperatureHeatProfile.currentProfile.size;
663 
664  // Check if we reached the end of the heating process
665  if (cOffset+1 >= mOffset) {
666  disableHeat();
667  } else {
668  // Calculate desired temperature
669  while (temperatureHeatProfile.currentProfile.time[cOffset+1] < cTimeStamp) {
670  if ((++cOffset) >= mOffset) {
671  disableHeat();
672  break;
673  }
674  }
675 
676  // Get the desired temperature & time in nearest defined points
677  uint16_t temperature1 = temperatureHeatProfile.currentProfile.data[cOffset];
678  uint16_t temperature2 = temperatureHeatProfile.currentProfile.data[cOffset+1];
679  uint16_t time1 = temperatureHeatProfile.currentProfile.time[cOffset];
680  uint16_t time2 = temperatureHeatProfile.currentProfile.time[cOffset+1];
681 
682  uint16_t dTtime = time2 - time1;
683  uint16_t dTtemp = temperature2 - temperature1;
684  uint16_t dcTtime = cTimeStamp - time1;
685 
686 
687 
688  // Calculate the desired temperature
689  double ratio = dcTtime/dTtime;
690  double diff = ratio * (dTtemp);
691  uint16_t desired_temp = temperature1 + (uint16_t)diff;
692 
693  // Get measured temperature
694  uint16_t current_temperature_X4 = readLeatestTemperatreMeasuremntX4();
695 
696  if (desired_temp < current_temperature_X4) {
697  SSR_OUTPUT_SetHigh();
698  }

700  }
701 
702  } else {
703  disableHeat();
704  }
705  }

711  addTask(IdleState,TranscieveNextionDATA);
