317 bool checkStartConditions() {
318 
319  if (ftdiStatus!=NORMAL_OPERATION)
320  return false; // Transmission in progress
321 
322  if (SENSOR_DATA_HANDLER.dataArrayQue[SENSOR_DATA_HANDLER.dataArrayStatus.currentData].s.oc)
323  return false; // Thermocouple is open
324 
325  if (SENSOR_DATA_HANDLER.dataArrayQue[SENSOR_DATA_HANDLER.dataArrayStatus.currentData].s.scg)
326  return false; // Thermocouple is short-circuited to GND
327 
328  if (SENSOR_DATA_HANDLER.dataArrayQue[SENSOR_DATA_HANDLER.dataArrayStatus.currentData].s.scv)
329  return false; // Thermocouple is short-circuited to VCC
330 
331  if (SENSOR_DATA_HANDLER.dataArrayQue[SENSOR_DATA_HANDLER.dataArrayStatus.currentData].s.internal_temperature_data < INTERNAL_MAX_TEMPERATURE)
332  return false; // Inernal maximum temperature exceeded
333 
334  if (SENSOR_DATA_HANDLER.dataArrayQue[SENSOR_DATA_HANDLER.dataArrayStatus.currentData].s.thermocouple_temperature_data < THERMOCOUPLE_MAX_TEMPERATURE)
335  return false; // Thermocouple maximum temperature exceeded
336 
337  return true;
